117:   /**
118:    * Grunt work to assign users
119:    *
120:    * It'd be best to run `assignmentRun()` as that method automatically detects errors
121:    * and fixes them. This is a helper processor.
122:    * 
123:    * @return void
124:    */
125:   public function runAssignment()
126:   {
127:     if (count($this->roles) == 0)
128:       throw new AllocatorException('Roles are not defined for allocation.');
129: 
130:     if (count($this->pools) == 0)
131:       throw new AllocatorException('Pools are not defined for allocation.');
132:     
133:     // Reset it
134:     $this->resetWorkflows();
135: 
136:     if (count($this->workflows) == 0)
137:       throw new AllocatorException('No workflows to allocate to.');
138: 
139:     // Now let's find the assignes
140:     foreach($this->roles as $role_id => $role_data) :
141:       $rolePool = $this->getPool($role_data['rules']['pool']['name'])->all();
142: 
143:       // Let's keep this very random
144:       $rolePool = shuffle_assoc($rolePool);
145: 
146:       // Add it to a queue
147:       $this->roles_queue[$role_id] = $rolePool;
148:     endforeach;
149: 
150:     // Go though the workflows
151:     foreach($this->workflows as $workflow_id => $workflow) :
152:       // Loop though each role inside of the workflow
153:       // 
154:       // Loop though all the users in the queue
155:       // 
156:       // Can join: assign and remove from queue
157:       // Can't join: point to next user in queue
158:       foreach($workflow as $role_id => $ignore) :
159:         // Just check if it's already assigned to be sure
160:         if ($ignore !== NULL)
161:           throw new AllocatorException(sprintf('Workflow role %d is already assigned to %d', $role_id, $ignore));
162:         
163:         $currentRole = $this->roles[$role_id];
164: 
165:         // If they aren't pulling from a list, they're going to be taking random items from a list
166:         if (! $currentRole['rules']['pool']['pull after']) :
167:           $this->roles_queue[$role_id] = shuffle_assoc($this->roles_queue[$role_id]);
168:         endif;
169: 
170:         // See if the task instance has a workflow alias
171:         if ($this->workflowTaskHasAlias($role_id, $currentRole, $this->workflows[$workflow_id]))
172:         {
173:           // Assign the user based upon the task alias
174:           $this->workflows[$workflow_id][$role_id] = $this->assignTaskAlias($role_id, $currentRole, $this->workflows[$workflow_id]);
175:         } else {
176:           // Start from the beginning of the queue
177:           foreach($this->roles_queue[$role_id] as $queue_id => $user) :
178:             // They're not a match -- skip to the next user in queue
179:             if ($this->canEnterWorkflow($user, $this->workflows[$workflow_id], $role_id))
180:             {
181:               $this->workflows[$workflow_id][$role_id] = $user;
182: 
183:               // Should they be removed from the pool?
184:               if ($currentRole['rules']['pool']['pull after'])
185:                 unset($this->roles_queue[$role_id][$queue_id]);
186: 
187:               break;
188:             }
189:           endforeach;
190:         }
191:         endforeach;
192:       endforeach;
193:   }

195:   /**
196:    * Identify if a user can enter a specific workflow
197:    *
198:    * Helper function to see if a user is already in a
199:    * workflow (cannot join then).
200:    * 
201:    * @param SectionUser User Object
202:    * @param array Workflow to check for entry
203:    * @param integer The Role ID to check
204:    * @return bool
205:    */
206:   protected function canEnterWorkflow($user, $workflow, $role_id)
207:   {
208:     $role = $this->roles[$role_id];
209: 
210:     foreach($workflow as $workflow_role_id => $assigne)
211:     {
212:       // Workflow Instance not assigned
213:       if ($assigne == NULL)
214:         continue;
215: 
216:       if ((int) $assigne->user_id == (int) $user->user_id)
217:       {
218:         // They've got a match! Now let's see if it's not an alias
219:         $workflowRole = $this->roles[$workflow_role_id];
220: 
221:         // The role they're trying to enter is an alias to a role that was already taken by
222:         // the user. A false positive.
223:         if ( isset($workflowRole['rules']['user alias']) AND $workflowRole['rules']['user alias'] == $role['name'])
224:           continue;
225:         else
226:           return FALSE;
227:       }
228:     }
229:     return TRUE;
230:   }

255:    * Assign the user based upon the task alias

275:   /**
276:    * Helper function to determine the workflow task alias
277:    *
278:    * @access protected
279:    * @return array|bool
280:    */
281:   protected function findWorkflowTaskAlias($role_id, &$workflow)
282:   {
283:     $role = $this->roles[$role_id];
284: 
285:     if (! isset($role['rules']['user alias'])) return false;
286: 
287:     // It has one -- let's find the other task instance
288:     $aliasRole = $role['rules']['user alias'];
289: 
290:     foreach ($this->roles as $sub_role_id => $sub_role_data)
291:     {
292:       if ($sub_role_data['name'] !== $aliasRole)
293:         continue;
294: 
295:       // Check for a double alias!
296:       // This is set on the instance that has a count > 1
297:       if (
298:         isset( $role['rules']['user alias all types'] )
299:       AND
300:         $role['rules']['user alias all types']
301:       AND
302:         isset ($role['rules']['count'])
303:       AND
304:         (int) $role['rules']['count'] > 1
305:       ) :
306:         // They have double alias protection on
307:         // Now we have to find the other instance of this role.
308:         $count = $role['rules']['count']-1;
309: 
310:         // Go up and down the workflow in the amount of instances they
311:         // have minus 1.
312:         
313:         // Search DOWN
314:         for ($i = $role_id-1; $i > $role_id-1-$count; $i--)
315:         {
316:           // Gone too far!
317:           if ($i < 0) break;
318: 
319:           // This instance isn't assigned
320:           if ($workflow[$i] !== NULL AND $this->roles[$i]['name'] == $role['name'])
321:             return FALSE;
322:         }
323: 
324:         // Search UP
325:         for ($i = $role_id+1; $i < $role_id+1+$count; $i++)
326:         {
327:           // No longer found.
328:           if (! isset($workflow[$i]))
329:             break;
330: 
331:           // This instance isn't assigned
332:           if ($workflow[$i] !== NULL AND $this->roles[$i]['name'] == $role['name'])
333:             return FALSE;
334:         }
335:       endif;
336: 
337:       // We've got a match!
338:       // Check to make sure we're not gonna 500
339:       if ($sub_role_id == $role_id)
340:         throw new AllocatorException(sprintf('Alias role ID is the same as parent role ID: %d %d', $aliasRoleId, $role_id));
341: 
342:       return [
343:         'role id' => $sub_role_id,
344:         'role data' => $sub_role_data,
345:       ];
346:       break;
347:     }
348: 
349:     // No match found.
350:     return FALSE;
351:   }

391:    * @return array

393:   public function emptyWorkflow($workflow_id)
394:   {
395:     // Let's get the tasks for this workflow
396:     $tasks = WorkflowTask::where('workflow_id', '=', $workflow_id)
397:       ->get();
398: 
399:     // Setup the instance storage for this workflow
400:     $this->taskInstanceStorage[$workflow_id] = $usedInstances = [];
401: 
402:     $i = [];
403:     foreach($this->roles as $role_id => $role_data) :
404:       // Determine which task instance this is
405:       $taskInstanceId = 0;
406:       foreach ($tasks as $task) :
407:         // It's a match
408:         if ($task->type == $role_data['name'] AND ! in_array($task->task_id, $usedInstances))
409:         {
410:           $taskInstanceId = $task->task_id;
411:           $usedInstances[] = $task->task_id;
412:           break;
413:         }
414:       endforeach;
415: 
416:       if ($taskInstanceId == 0)
417:         throw new AllocatorException(
418:           sprintf('Unknown task instance id to assign for role %s of workflow %d', $role_data['name'], $workflow_id)
419:         );
420:       else
421:         $this->taskInstanceStorage[$workflow_id][$role_id] = $taskInstanceId;
422: 
423:       // Add this to the tempory storage
424:       $i[$role_id] = NULL;
